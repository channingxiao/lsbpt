116 void imageblock::processing()
117 {
118 
119 #pragma omp critical
120  {
121  //load color intensity info
122  this->loadBlock();
123  }

125  //load probability

127  this->loadProbabilities();
128  }
129  this->InitializeRag();

131 
132 
133 
134  this->selectedSegmentationNodes.push_back( root );
135  for(int i=0; i < this->lamda.size(); i++){
136  std::vector<Node*> tempSelected;
137  for(int j=0; j<this->selectedSegmentationNodes.size(); j++){
138  double cost;
139  std::vector<Node*> temp;
140  io::bestSegmentation(this->selectedSegmentationNodes[j], temp, cost, this->lamda[i], this->useStdCut);
141  tempSelected.insert( tempSelected.begin(), temp.begin(), temp.end());
142  }
143  this->selectedSegmentationNodes.clear();
144  this->selectedSegmentationNodes = tempSelected;
145  tempSelected.clear();
146 
147 #pragma omp critical
148  {
149  this->saveToDisk(i);
150  }
151  }
152 
153 
154  this->root->releaseTree();

179 void imageblock::loadProbabilities( )
180 {
187  int n_items = SizeX*SizeY*nClasses;
188  std::vector<double> probData(n_items);
189  std::ifstream dataFile;
190 
191  dataFile.open(this->fileNameProbabilities.c_str(), std::ios_base::in | std::ios_base::binary);
192  dataFile.read(reinterpret_cast<char*>(&probData[0]), n_items*sizeof(double));
193  dataFile.close();
194 
195  std::vector<std::vector<std::vector<double> > > probabilities(SizeY,std::vector< std::vector <double> >(SizeX, std::vector<double>(nClasses)) );
196  for (size_t i=0;i<probabilities.size();i++)
197  for (size_t j=0;j<probabilities[0].size();j++){
198  for (size_t p=0;p<probabilities[0][0].size();p++){
199 
200  probabilities[i][j][p] = probData[i*SizeX*nClasses + j*nClasses + p];
201  }
202  }
203 
204  this->probabilities = probabilities;
205 
206  std::cout << "Pribability loaded" << endl;
207  std::cout << "Probability file size is : " << probabilities.size() << " " << probabilities[0].size() << endl;
208 
209 }

214 void imageblock::saveToDisk( int level )
215 {
227  string fileNameSegmentation = this->directoryOutput + "/segmentation" + std::to_string(level) + ".tif";
228  string fileNameSegmentationCSV = this->directoryOutput + "/segmentation" + std::to_string(level) + ".csv";
229 
230  GDALDataset * classDataset;
231  if( this->ifAddProbabilityModel ){
232  std::string fileNameClassMap = this->directoryOutput + "/classMap" + std::to_string(level) + ".tif";
233  classDataset = (GDALDataset *) GDALOpen( fileNameClassMap.c_str(), GA_Update);
234  if( classDataset == NULL )
235  {
236  std::cout<< "class map file open failed" <<endl;
237  exit(1);
238  }
239  }
240 
241 
242  GDALDataset * poDataset;
243  GDALAllRegister();
244  poDataset = (GDALDataset *) GDALOpen( fileNameSegmentation.c_str(), GA_Update);
245  if( poDataset == NULL )
246  {
247  std::cout<< "Segmentation file open failed" <<endl;
248  exit(1);
249  }
250 
251  ofstream fout;
252  fout.open((fileNameSegmentationCSV).c_str(), ios::app);
253 
254  ofstream foutBoarder;
255  std::string boraderFileName = "i" + std::to_string(this->blockPosition[0]) + "j" + std::to_string(this->blockPosition[1]);
256  foutBoarder.open((boraderFileName).c_str(), ios::app);
257 
258 
259  for(size_t i=0; i< selectedSegmentationNodes.size(); i++) {
260 
261  if(selectedSegmentationNodes[i]->getBoarderIndexX().size() == 0
262  && selectedSegmentationNodes[i]->getBoarderIndexY().size() == 0) {
263  // if not touching any tiling boundary
264  int *color = new int[3];
265  for(int j=0; j <3 ; j++){
266  if(this->ifPaintRandomColor) {
267  color[j] = std::rand()%255;
268  }else{
269  color[j] = selectedSegmentationNodes[i]->getMeanSpectrumAt(j);
270  }
271  }
272 
273  io::writeNodeToDisk(selectedSegmentationNodes[i], color, poDataset, this->globalStartX, this->globalStartY );
274  io::writeNodeToCSV(selectedSegmentationNodes[i], rand()%135797531, fout);
275  if( this->ifAddProbabilityModel ){
276  io::writeClassMap( selectedSegmentationNodes[i], selectedSegmentationNodes[i]->getClassLabel() , classDataset, this->globalStartX, this->globalStartY );
277  }
278  delete[] color;
279 
280  } else if( level==0){
281  // if touching some tiling boundaries
282  io::writeBoarderNodeToCSV( selectedSegmentationNodes[i] , rand()%135797531, foutBoarder, this->ifAddProbabilityModel);
283  }
284 
285  }
286 
287 
288 
289 
290 
291  delete poDataset;
292  if( this->ifAddProbabilityModel ) delete classDataset; fout.close();
293  foutBoarder.close();
294 }
