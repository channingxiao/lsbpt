34 void ImageCropper::on_cut_clicked()
35 {
36  //parse bandMap, if input is 0, them select all bands
37  std::vector<int> bandMap;
38  if( !this->ui->bandMap->text().isEmpty() ){
39  std::string bandMapString = this->ui->bandMap->text().toStdString();
40  stringstream bandMapStream(bandMapString);
41  int i;
42  while( bandMapStream >> i) {
43  bandMap.push_back( i );
44  if( bandMapStream.peek() == ',' | bandMapStream.peek() == ' ')
45  bandMapStream.ignore();
46  }
47 
48  }
49 
50  this->imageCutSizeX = this->ui->width->value();
51  this->imageCutSizeY = this->ui->height->value();
52  this->startPointX = this->ui->startX->value();
53  this->startPointY = this->ui->startY->value();
54  this->nBlocksX = this->ui->nBlocksX->value();
55  this->nBlocksY = this->ui->nBlocksY->value();
56  this->ifDeleteNoise = this->ui->ifDeleteNoise->isChecked();
57  this->nStdLeft = this->ui->nStdLeft->value();
58  this->nStdRight = this->ui->nStdRight->value();
59  this->ifRescale = this->ui->ifRescale->isChecked();
60  this->scaledMin = this->ui->scaledMin->value();
61  this->scaledMax = this->ui->scaledMax->value();

63  //qDebug() << this->scaledMin <<" " << this->scaledMax << endl;
64 
65  this->blockSizeCalculator_evenSplite();

67 
68  this->cropImage(bandMap);

143 void ImageCropper::blockSizeCalculator_evenSplite()
144 { // this function evenly splite the image
145 
146  blockSizeW.clear();
147  blockSizeH.clear();
148 
149  int eachSizeX = imageCutSizeX/nBlocksX;
150  int eachSizeY = imageCutSizeY/nBlocksY;
151 
152  for(int i= 0;i< nBlocksX-1; i++ )
153  blockSizeW.push_back(eachSizeX);
154  blockSizeW.push_back(imageCutSizeX - eachSizeX*(nBlocksX-1));
155 
156  for(int j=0; j<nBlocksY-1; j++)
157  blockSizeH.push_back(eachSizeY);
158  blockSizeH.push_back(imageCutSizeY- eachSizeY*(nBlocksY-1));
159 
160  return;
161 }

175  blocksStartPointX.push_back(blocksStartPointX[i-1] + blockSizeW[i-1]);

177  for(int j=1; j < nBlocksY; j++)
178  blocksStartPointY.push_back(blocksStartPointY[j-1]+ blockSizeH[j-1]);
179 
180  return;

185 void ImageCropper::cropImage(std::vector<int> &bandIndex)
186 {
187 
188  // The band index start from 1
189 
190 
191  GDALAllRegister();
192  GDALDataset *poDataset = (GDALDataset *) GDALOpen( this->imageFileName.c_str(), GA_ReadOnly ); // open file
193  //int BandCount = poDataset->GetRasterCount();
194 
195  const char *pszFormat = "GTiff";
196  GDALDriver *poDriver = GetGDALDriverManager()->GetDriverByName( pszFormat );
197  char **papszOptions = NULL;
198  papszOptions = CSLSetNameValue( papszOptions, "TILED", "NO" );
199  papszOptions = CSLSetNameValue( papszOptions,"BIGTIFF", "IF_NEEDED" );
200 
201 
202  std::vector<double> rasterMin;
203  std::vector<double> rasterMax;
204  std::vector<double> rasterMean;
205  std::vector<double> rasterStdDev;
206  // for preprocessing
207 
208  std::vector<double> orderMin;
209  std::vector<double> orderMax;
210 
211  //if the bandIndex is empty, do for all bands
212  if(bandIndex.size() == 0) {
213  for(int i=1; i < poDataset->GetRasterCount()+1; i++)
214  bandIndex.push_back(i);
215 
216  }
217 
218 
219 
220  for(size_t i=0; i < bandIndex.size(); i++) {
221 
222  GDALRasterBand * rasterBand = poDataset->GetRasterBand( bandIndex[i] );
223  int bApproxOK = 0;
224  int bForce = 1;
225  double min=0;
226  double max=0;
227  double mean=0;
228  double stdDev=0;
229  rasterBand->GetStatistics(bApproxOK, bForce, &min, &max, &mean, &stdDev);
230  rasterMin.push_back(min);
231  rasterMax.push_back(max);
232  rasterMean.push_back(mean);
233  rasterStdDev.push_back(stdDev);
234 
235  orderMin.push_back( rasterMean[i] - nStdLeft*rasterStdDev[i] );
236  orderMax.push_back( rasterMean[i] + nStdRight*rasterStdDev[i] );
237 
238  }
239 
240  /*
241 
242  orderMin[0] = 0;
243  orderMin[1] = 0;
244  orderMin[2] = 0;
245  orderMax[0] = 769;
246  orderMax[1] = 710;
247  orderMax[2] = 706;
248 
249 
250  */
251 
252 
253 
254  for(int i = 0; i < this->nBlocksX; i++)
255  for(int j = 0; j < this->nBlocksY; j++) {
256 
257  std::string fileName_out = this->directoryName + "/" + to_string(i/10) + to_string( i%10 ) +to_string(j/10) + to_string(j%10) + ".tif";
258 
259  int width = this->blockSizeW[i];
260  int height = this->blockSizeH[j];
261  int startX = this->blocksStartPointX[i];
262  int startY = this->blocksStartPointY[j];
263  int BandCount = bandIndex.size();
264 
265  GDALDataset *data_out;
266  data_out= poDriver->Create(fileName_out.c_str(),width,height,BandCount,GDT_Byte,papszOptions);
267 
268  double *buffer= new double [width*height*BandCount];
269  int *panBandMap = new int[BandCount];
270  for(int k=0; k < BandCount; k++) panBandMap[k] = bandIndex[k];
271  poDataset->RasterIO( GF_Read, startX, startY,width,height,
272  buffer,width,height,GDT_Float64,BandCount,panBandMap,0,0,0);
273 
274  if( this->ifDeleteNoise ) {
275  clip<double>(buffer, orderMin, orderMax,
276  BandCount, width*height);
277  }
278 
279  // data_out->RasterIO( GF_Write, 0, 0,width,height,
280  // buffer,width,height,GDT_Float64,BandCount,0,0,0,0 );
281 
282  uint8_t *buffer_int = new uint8_t[width*height*BandCount];
283  for(int k=0; k < width*height*BandCount; k++) buffer_int[k] = static_cast<uint8_t>(buffer[k]);
284 
285  data_out->RasterIO( GF_Write, 0, 0,width,height,
286  buffer_int,width,height,GDT_Byte,BandCount,0,0,0,0 );
287 
288  delete[] buffer;
289  delete[] buffer_int;
290  delete data_out;
291 
292  }
293 
294  delete poDataset;
295  return;
296 
297 }
