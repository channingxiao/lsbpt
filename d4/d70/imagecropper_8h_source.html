49  template<class T>
50  void clip(T *buffer, const std::vector<double> &min, const std::vector<double> &max,
51  int numBand, size_t numPixel)
52  {
53  //check the input
54  if(min.size() != max.size()) {
55  exit(1);
56  }else if( min.size() != numBand ){
57  exit(1);
58  }
59 
60  for(int i=0; i<numBand; i++) {
61  for(int k = numPixel*i; k< numPixel*(i+1); k++){
62 
63  if( buffer[k] > max[i]) {
64  buffer[k]= max[i];
65  } else if( buffer[k] <min[i]) {
66  buffer[k] = min[i];
67  }
68 
69  if( ifRescale ) {
70  rescale<T> (buffer[k], min[i], max[i], scaledMin , scaledMax );
71  }
72 
73  }
74  }
75 
76  return;
